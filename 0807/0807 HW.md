### 1974. 스도쿠 검증

```python
import copy
T = int(input())

def numbers(a):
    b = copy.deepcopy(a)
    for i in range(len(b)-1):
        minnum = i
        for j in range(i+1, len(b)):
            if b[minnum] > b[j]:
                minnum = j
        b[minnum], b[i] = b[i], b[minnum]
    # b.sort()
    for i in range(len(b)):
        if i+1 != b[i]:
            return 'False'
    return True


for z in range(T):
    s_li = [list(map(int, input().split())) for _ in range(9)]
    result = 1

    for i in range(9):
        if numbers(s_li[i]) == 'False':
            result = 0

    for i in range(9):
        c_li = []
        for j in range(9):
            c_li += [s_li[j][i]]
        if numbers(c_li) == 'False':
            result = 0

    for i in range(0, 9, 3):
        for j in range(0, 9, 3):
            n_li = []
            for a in range(i, i+3):
                for b in range(j, j+3):
                    n_li += [s_li[a][b]]
            if numbers(n_li) == 'False':
                result = 0

    print('#{} {}'.format(z+1, result))
```

### 1258. 행렬찾기

```python
T = int(input())
for z in range(T):
    n = int(input())
    matrix = [list(map(int, input().split())) for _ in range(n)]

    area = []
    for i in range(n):
        for j in range(n):
            if matrix[i][j] != 0:
                x = 0
                for k in range(j, n):
                    if matrix[i][k]:
                        x += 1
                    else:
                        break
                y = 0
                for k in range(i, n):
                    if matrix[k][j]:
                        y += 1
                    else:
                        break

                for k in range(i, i + y):
                    for l in range(j, j + x):
                        matrix[k][l] = 0

                area.append([y, x, y * x])

    for i in range(len(area)):
        minnum = i
        for j in range(i+1, len(area)):
            if area[minnum][2] > area[j][2]:
                minnum = j
        area[minnum], area[i] = area[i], area[minnum]

    print('#{} {}'.format(z+1, len(area)), end=" ")
    for i in range(len(area)):
        for j in range(2):
            print('{}'.format(area[i][j]), end =" ")
    print()
```
